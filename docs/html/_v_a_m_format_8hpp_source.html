60 // Validation helpers
61 [[nodiscard]] bool IsValid() const
62 {
63 return memcmp(magic, VAM_MAGIC, 4) == 0 && version == VAM_VERSION;
64 }
65
66 [[nodiscard]] bool IsCompressed() const
67 {
68 return (flags & static_cast<uint32_t>(VAMFlags::Compressed)) != 0;
69 }
70

76 void SetCompressionRatio(const float ratio)
