224 FixedStorage<T, CAPACITY>::~FixedStorage()
225 {
226 // Release all allocated objects to ensure proper destruction
227 for (size_t i = 0; i < CAPACITY; ++i)

229 if (auto& slot = m_Slots[i]; slot.inUse.load(std::memory_order_acquire))

238 template <typename T, size_t CAPACITY>
239 template <typename... Args>
240 typename FixedStorage<T, CAPACITY>::HandleType FixedStorage<T, CAPACITY>::Allocate(
241 Args&&... args)
242 {
243 // Find and claim a free slot
244 size_t slotIndex = FindAndClaimFreeSlot();
245 if (slotIndex >= CAPACITY)
246 {
247 VA_ENGINE_WARN(
248 "[FixedStorage<{}>] The storage is full ({}/{} slots used).",
249 typeid(T).name(),
250 GetUsedSlots(),
251 CAPACITY);
252
253 return HandleType::Invalid();
254 }
255
256 auto& slot = m_Slots[slotIndex];
257
258 // Increment generation for new allocation
259 uint32_t generation = slot.generation.fetch_add(1, std::memory_order_acq_rel) + 1;
260
261 try
262 {
263 // Construct an object in-place with perfect forwarding
264 T* object = slot.GetObject();
265 new(object) T(std::forward<Args>(args)...);
266
267 // Update statistics
268 m_UsedCount.fetch_add(1, std::memory_order_relaxed);
269
270 // Update hint for next allocation
271 m_NextSlot.store((slotIndex + 1) % CAPACITY, std::memory_order_relaxed);
272
273 return HandleType(static_cast<uint32_t>(slotIndex), generation);
274 }
275 catch (...)
276 {
277 // Construction failed - release claimed slot
278 slot.generation.fetch_sub(1, std::memory_order_acq_rel);
279 slot.inUse.store(false, std::memory_order_release);
280
281 VA_ENGINE_ERROR(
282 "[FixedStorage<{}>] Failed to construct object in slot {}.",
283 typeid(T).name(),
284 slotIndex);
285
286 return HandleType::Invalid();
287 }
288 }

345 template <typename T, size_t CAPACITY>
346 size_t FixedStorage<T, CAPACITY>::FindAndClaimFreeSlot()
347 {
348 const uint32_t startSlot = m_NextSlot.load(std::memory_order_relaxed);
349
350 // Search for a free slot with wraparound
351 for (size_t attempts = 0; attempts < CAPACITY; ++attempts)
352 {
353 size_t slotIndex = (startSlot + attempts) % CAPACITY;
354 auto& slot = m_Slots[slotIndex];
355
356 // Try to claim this slot
357 if (auto expected = false; slot.inUse.compare_exchange_weak(
358 expected,
359 true,
360 std::memory_order_acquire,
361 std::memory_order_relaxed))
362 {
363 return slotIndex;
364 }
365 }
366
367 return CAPACITY; // No free slots found - return an invalid handle
368 }

370 template <typename T, size_t CAPACITY>
371 bool FixedStorage<T, CAPACITY>::ValidateHandle(HandleType handle, size_t& slotIndex) const
372 {
373 if (!handle.IsValid()) return false;
374
375 slotIndex = handle.GetIndex();
376 if (slotIndex >= CAPACITY) return false;
377
378 const auto& slot = m_Slots[slotIndex];
379
380 if (!slot.inUse.load(std::memory_order_acquire)) return false;
381
382 // Check generation to prevent ABA problems
383 uint32_t curGen = slot.generation.load(std::memory_order_relaxed);
384 if (curGen != handle.GetGeneration()) return false;
385
386 return true;
387 }
